python部分

网页解析

虽然之前也接触过一些相关知识，但浅尝辄止。demo主要使用bs4、xpath，上线使用bs4。
碰到的主要问题：

  • wiki的表格存在行列无法完全匹配的问题
  • 舰船名3个到5个不等(舰船番号、舰船名、游戏名、英译、日译)
  • 技能描述过长、大世界技能、改造技能
  • 台词过长、合并（正则处理）
  • 反爬机制（虽然实际上只需要随机headers+sleep+retry ╰(￣ω￣ｏ))
后记
  • 后续清理死代码，简直无地自容(っ °Д °;)っ
  • 正则、bs4虽然还是处于离了google就活不下去的阶段，不过总算是真正开始动手了

数据库

表设计
表数据主键
name_urls舰船链接有
profile舰船基本介绍(编号、入港时间、相关活动、获取途径)有
portrait角色介绍有
names舰船名有
equipments突破后装备加成有
equipments_re改造后装备加成有
meta_strmeta阵营角色强化数据有
per_rank各属性等级有
per_rank_re改造后各属性等级有
performance各属性数值有
performance_re改造后各属性数值有
performance_su潜艇各属性数值有
retrofit_cost改造需求无
skills技能及具体描述有
skills_re改造后技能有
tech各阶段科技点加成有
tech_bt各阶突破后舰船加成有
tech_bt_dev科研开发等级加成无
tech_bt_fit天运加成无
words_region原生台词无
words_skins皮肤台词无
words_valentine情人节台词无

Python创建表

使用pymysql创建表

import pymysql
# mysql -e 'create database azurelane  DEFAULT CHARACTER SET utf8 DEFAULT COLLATE utf8_general_ci;' -u root -p

conn = pymysql.connect(host = 'localhost', 
                             user = 'root' ,
                             password = 'passwd', 
                             cursorclass =  pymysql.cursors.DictCursor
                             # db = 'mysql'
                             )
cur = conn.cursor()
cur.execute("USE azurelane;")
# create_database
## name_urls
cur.execute("CREATE TABLE name_url  (name char(15) , url char(150) )character set utf8mb4;")

## names 
cur.execute("CREATE TABLE names  (No int(5), Number char(50) , zh char(50)  ,game  char(50) ,en char(50) ,jp char(50) )character set utf8mb4;")
## portrait
cur.execute("CREATE TABLE  portrait (No int(5), cv char(100), mom char(50) ,identity char(150), disposition char(50) , keywords char(100) , possession char(100), color_hair char(100), color_eyes char(100), cute char(255) )character set utf8mb4;")

##  No && profiles
cur.execute("CREATE TABLE  profile (No int(5) ,Name char(50), Type char(4) , Belong char(10) , Rarity char(5) , Date char(20) ,Activity char(200) ,Time char(20) ,Common_drop char(255) ,Archives_drop char(155) ,Activity_drop char(255) ,Other_access char(255) ,Value char(25) ,Retired char(20) ,Shelling_exp   int(4) ,Torpedo_exp  int(4) ,Aviation_exp  int(4) ,Filling_exp int(4) )character set utf8mb4;")

##  No && region_words
cur.execute("CREATE TABLE words_region  (No int(5), word_name char(20) , word_content char(255) )character set utf8mb4;")

##  No && skin_words
cur.execute("CREATE TABLE words_skins  (No int(5) ,skin char(40) ,  word_name char(10) ,word_content char(255) )character set utf8mb4;")

##  No && valentine_words
cur.execute("CREATE TABLE words_valentine (No int(5), year char(5) , word_content  char(255) )character set utf8mb4;")

##  No && skill
cur.execute("CREATE TABLE  skills (No int(5) , skill_1 char(50) ,skill_1_desc char(255) ,skill_ex_1 char(50) ,skill_ex_1_desc char(255) ,skill_2 char(50) ,skill_2_desc char(255) ,skill_ex_2 char(50) ,skill_ex_2_desc char(255) ,skill_3 char(50) ,skill_3_desc char(255) ,skill_ex_3 char(50) ,skill_ex_3_desc char(255) , skill_4 char(50) ,skill_4_desc char(255) ,skill_ex_4 char(50) ,skill_ex_4_desc char(255) )character set utf8mb4;")

cur.execute("CREATE TABLE skills_re ( No int(5), re_skill_1 char(50) , re_skill_1_desc char(255) ,re_skill_2 char(50) , re_skill_2_desc char(255), re_skill_3 char(50) , re_skill_3_desc char(255), re_skill_4 char(50) , re_skill_4_desc char(255))character set utf8mb4;")
## retrofit_cost
# ['改造项目', '项目属性', '需求图纸', '需求物资', '需求等级', '需求星级']
cur.execute("CREATE TABLE  retrofit_cost (No int(5), project char(20), attribute char(255) , drawing char(100) , cost char(30) , level char(2), star_level char(2))character set utf8mb4;")

## No && tech-breakthrough
# 科技加成
# 天运
cur.execute("CREATE TABLE  tech_bt_dev (No int(5), dev_level char(5),dev_desc char(100) ) character set utf8mb4;")

cur.execute("CREATE TABLE  tech_bt_fit (No int(5), fit_level char(5) , fit_desc char(255)) character set utf8mb4;")

cur.execute("CREATE TABLE  tech_bt (No int(5),bt_1 char(80) , bt_2 char(80), bt_3 char(80) ) character set utf8mb4;")

# 属性加成&&科技点
cur.execute("CREATE TABLE tech (No int(5), get_attrs char(15) , get_num int(2) , bt_attrs char(15) , bt_num int(2), 120_attrs char(15), 120_num int(2)) character set utf8mb4;")

## rank
cur.execute("CREATE TABLE per_rank (No int(4),hp_rank char(1) , air_defense_rank char(1) , maneuvering_rank char(1) , aviation_rank char(1) , torpedo_rank char(1) ,shelling_rank char(1)) character set utf8mb4;")

cur.execute("CREATE TABLE per_rank_re (No int(4),hp_rank_ex char(1) , air_defense_rank_ex char(1) , maneuvering_rank_ex char(1) , aviation_rank_ex char(1) , torpedo_rank_ex char(1) , shelling_rank_ex char(1) ) character set utf8mb4;")
 
# meta
cur.execute("CREATE TABLE  meta_str (No int(5), shelling char(3),hp char(5),torpedo char(3),air_defense char(3),aviation  char(3),hit char(3),filling char(3), maneuvering char(3) )character set utf8mb4;")
## performance
cur.execute("CREATE TABLE performance (No int(5),armor_type char(4), lucky char(6), hp_num_o char(5),filling_num_o char(3), shelling_num_o char(3),maneuvering_num_o char(3), torpedo_num_o char(3),air_defense_num_o char(3),aviation_num_o  char(3),hit_num_o char(3), anti_submarine_num_o char(3), cost_num_o char(2),speed_o char(3), hp_num char(5),filling_num char(3), shelling_num char(3),maneuvering_num char(3), torpedo_num char(3),air_defense_num char(3),aviation_num  char(3),hit_num char(3), anti_submarine_num char(3), cost_num char(2),speed char(3)) character set utf8mb4;")

## retrofit_performance_ex
cur.execute("CREATE TABLE performance_re (No int(5),armor_type char(4), lucky char(6),hp_num_re char(5),filling_num_re char(3), shelling_num_re char(3),maneuvering_num_re char(3), torpedo_num_re char(3),air_defense_num_re char(3),aviation_num_re  char(3),hit_num_re char(3), anti_submarine_num_re char(3), cost_num_re char(2),speed_re char(3)) character set utf8mb4;")
# submarine
cur.execute("CREATE TABLE performance_su (No int(5),armor_type char(4), lucky char(6),oxygen char(10),  submarine char(3) ,hp_o char(5),filling_o char(3), shelling_o char(3),maneuvering_o char(3), torpedo_o char(3),air_defense_o char(3),aviation_o char(3),hit_o char(3), cost_o char(2),speed_o char(3),hp char(5),filling char(3), shelling char(3),maneuvering char(3), torpedo char(3),air_defense char(3),aviation char(3),hit char(3), cost char(2),speed char(3)) character set utf8mb4;")

## No && equipments
cur.execute("CREATE TABLE equipments  (No int(5), dev_eq_1 char(50) ,dev_eq_1_efficient char(4) ,dev_eq_1_num char(5) , dev_eq_1_pre_num char(1) ,  dev_eq_2 char(50),dev_eq_2_efficient char(4),dev_eq_2_num char(1) , dev_eq_2_pre_num char(5) , dev_eq_3 char(50),dev_eq_3_efficient char(4),dev_eq_3_num char(1) , dev_eq_3_pre_num char(5) ,desc_add char(40),eq_desc char(150))character set utf8mb4;")

cur.execute("CREATE TABLE equipments_re (No int(5),dev_eq_1_re char(50) ,dev_eq_1_efficient_re char(4) ,dev_eq_1_num_re char(1) , dev_eq_1_pre_num_re char(5) ,dev_eq_2_re char(50),dev_eq_2_efficient_re char(4), dev_eq_2_num_re char(1) , dev_eq_2_pre_num_re char(5) , dev_eq_3_re char(50) ,dev_eq_3_efficient_re char(4),dev_eq_3_num_re char(5) , dev_eq_3_pre_num_re char(1) ,eq_desc char(150)) character set utf8mb4; ")
Mysql联合查询
  1. 为name_url表插入No并设置为主键
    1. 从profile表获取名字相等的记录并返回No
      1.
    2. 在python代码中依次插入No
  2. 查询
    1. 部分表格（皮肤台词等）每一行对应一个皮肤的一句台词，无法为每一行设置主键
    2. 查询(多表查询)
      1. 舰船名——>No——>皮肤——>台词（台词表格无主键）
      2. 舰船名(——>No)——>舰种、入港时间、阵营、获取途径、是否可改造、各属性等级、各属性数据（属性数据大部分有主键）
